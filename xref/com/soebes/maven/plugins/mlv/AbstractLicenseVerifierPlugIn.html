160     /**
161      * Get all their dependencies and put the information
162      * into the intermediate list of licenses.
163      *
164      * @param depArtifacts
165      * @throws MojoExecutionException
166      */
167     protected void getDependArtifacts(Set<?> depArtifacts)
168             throws MojoExecutionException {
169 
170         PatternExcludeFilter patternExcludeFilter = new PatternExcludeFilter();
171         ArtifactFilter filter = patternExcludeFilter.createFilter(excludes);
172 
173         for (Iterator depArtIter = depArtifacts.iterator(); depArtIter.hasNext(); ) {
174            Artifact depArt = (Artifact) depArtIter.next();
175 
176            if (!filter.include(depArt)) {
177                if (isVerbose()) {
178                    getLog().warn("The artifact: " + depArt.getId() + " has been execluded by the configuration.");
179                }
180                continue;
181            }
182 
183            LicenseInformation li = new LicenseInformation();
184 
185            // Here we access to the files so we can check if they contain LICENSE file etc. (check for JAR file reading? Other file types ? )
186 //           if (verbose) {
187 //               getLog().info("Artifact file:" + depArt.getFile().getAbsolutePath());
188 //           }
189 
190            //store the artifact about which the following license information
191            //will be extracted.
192            li.setArtifact(depArt);
193            MavenProject depProject = null;
194            try
195            {
196               depProject = projectBuilder.buildFromRepository(depArt, remoteRepositories, localRepository, true);
197            }
198            catch (ProjectBuildingException e)
199            {
200               throw new MojoExecutionException( "Unable to build project: " + depArt.getDependencyConflictId(), e );
201            }
202 
203            // depArt.getScope() => Can be use eventually
204            //Set the project of the current license information
205            li.setProject(depProject);
206 
207            //Add all licenses of the particular artifact to it's other informations.
208            List<?> licenses = depProject.getLicenses();
209            Iterator<?> licenseIter = licenses.iterator();
210            while (licenseIter.hasNext())
211            {
212               License license = (License) licenseIter.next();
213               li.addLicense(license);
214            }
215            licenseInformations.add(li);
216         }
217     }

227     /**
228      * This method will load the licenses.xml file.
229      *
230      * @throws MojoExecutionException
231      */
232     protected void loadLicensesFile() throws MojoExecutionException {
233         if (licenseFile == null)
234         {
235             //If no licenseFile configuration given we will end.
236             return;
237         }
238         try {
239             getLog().info("Loading " + licenseFile.getAbsolutePath() + " licenses file.");
240             licensesContainer = LicensesFile.getLicenses(licenseFile);
241             licenseValidator = new LicenseValidator(licensesContainer);
242             licenseValidator.setStrictChecking(stricktChecking);
243         } catch (IOException e) {
244             //Use the internal licenses.xml file.???
245             throw new MojoExecutionException(
246                 "The LicenseFile configuration is wrong, " +
247                 "cause we couldn't find the " + licenseFile.getAbsolutePath());
248         } catch (XmlPullParserException e) {
249             //Use the internal licenses.xml file.???
250             throw new MojoExecutionException(
251                 "The LicenseFile is wrong, " +
252                 "cause we couldn't read it " + e);
253         }
254     }
