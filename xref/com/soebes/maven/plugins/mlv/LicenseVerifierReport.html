195             //FIXME: Removed hard code scopes

201             sink.section1();
202             //FIXME: Removed hard code scopes

212     private boolean hasScope(String scope) {
213         boolean result = false;
214         for (LicenseInformation item : getLicenseInformations()) {
215             if (scope.equals(item.getArtifact().getScope())) {
216                 result = true;
217             }
218         }
219         return result;
220     }

282     private void doGenerateArtifactCategories(ResourceBundle bundle, Sink sink, String scope) {
283         sink.sectionTitle1();
284         sink.text("Artifact Categories (" + scope + ")");
285         sink.sectionTitle1_();
286 
287         if (hasValidEntries(scope)) {
288             doGenerateValidReport(bundle, sink, scope);
289         }
290         if (hasWarningEntries(scope)) {
291             doGenerateWarningReport(bundle, sink, scope);
292         }
293         if (hasInvalidEntries(scope)) {
294             doGenerateInvalidReport(bundle, sink, scope);
295         }
296         if (hasUnknownEntries(scope)) {
297             doGenerateUnknownReport(bundle, sink, scope);
298         }
299     }

301     private void doGenerateReportLicensesConfiguration(ResourceBundle bundle, Sink sink, String header, List<LicenseItem> licenseItems) {
302         sink.sectionTitle2();
303         sink.text(header);
304         sink.sectionTitle2_();
305 
306         sink.table();
307 
308         sink.tableRow();
309         headerCell(sink, "Id");
310         headerCell(sink, "Description");
311         headerCell(sink, "Names");
312         headerCell(sink, "URL's");
313         sink.tableRow_();
314 
315         for (LicenseItem item : licenseItems) {
316             sink.tableRow();
317             cell(sink, item.getId());
318             cell(sink, item.getDescription());

320             sink.tableCell();

322             sink.list();
323             for (String itemName : item.getNames()) {
324                 sink.listItem();

327             }
328             sink.list_();
329             sink.tableCell_();
330 
331             sink.tableCell();

365     private void doGenerateItemReport(ResourceBundle bundle, Sink sink) {
366         sink.sectionTitle1();
367         sink.text("Artifact License Categories");
368         sink.sectionTitle1_();
369 
370         sink.sectionTitle2();
371         sink.text("Overview");
372         sink.sectionTitle2_();
373 
374         sink.table();
375 
376         sink.tableRow();
377         headerCell(sink, "Id");
378         headerCell(sink, "Scope");
379         headerCell(sink, "Valid");
380         headerCell(sink, "Warning");
381         headerCell(sink, "Invalid");
382         headerCell(sink, "Unknown");
383         sink.tableRow_();
384 
385         for (LicenseInformation item : getLicenseInformations()) {
386             sink.tableRow();
387 
388             cell(sink, item.getArtifact().getId()); // 1st Row item artifactId
389 
390             boolean isValid = licenseValidator.isValid(item.getLicenses());
391             boolean isWarning = licenseValidator.isWarning(item.getLicenses());
392             boolean isInvalid = licenseValidator.isInvalid(item.getLicenses());
393             boolean isUnknown = licenseValidator.isUnknown(item.getLicenses());
394 
395             sink.tableCell();
396             sink.text(item.getArtifact().getScope());
397             sink.tableCell_();
398 
399             createFigure(sink, isValid);
400 
401             if (isWarning) {
402                 createFigure(sink, false);
403             } else {
404                 createEmptyCell(sink);
405             }
406 
407             if (isInvalid) {
408                 createFigure(sink, false);
409             } else {
410                 createEmptyCell(sink);
411             }
412 
413             if (isUnknown) {
414                 createFigure(sink, false);
415             } else {
416                 createEmptyCell(sink);
417             }
418 
419             sink.tableRow_();
420         }
421 
422         sink.table_();
423     }

616     public void generate(Sink sink, Locale locale) throws MavenReportException {
617         getLog().info("LicenseVerifierReport:generate()");
618         if (project.getArtifacts().isEmpty()) {
619             //If we have no artifacts at all.
620             doGenerateEmptyReport(getBundle(locale), sink);
621         } else {
622             try {
623                 loadLicensesFile();
624             } catch (MojoExecutionException e1) {
625                 throw new MavenReportException("Failure during load of the license.xml file", e1);
626             }
627 
628             //Get a set with all dependent artifacts incl.
629             //the transitive dependencies.
630             Set depArtifacts = project.getArtifacts();
631 
632             //Get all the informations about the licenses of the artifacts.
633             try {
634                 getDependArtifacts(depArtifacts);
635 
636                 //Sorting the artifacts...
637                 Collections.sort(getLicenseInformations(), new ArtifactComperator());
638                 doGenerateReport(getBundle(locale), sink);
639 
640             } catch (MojoExecutionException e) {
641                 getLog().error("Something wrong (BETTER MESSAGE): ", e);
642             }
643         }
644     }
