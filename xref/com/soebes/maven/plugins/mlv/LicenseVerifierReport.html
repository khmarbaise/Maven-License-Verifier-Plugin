152     private void doGenerateReport(ResourceBundle bundle, Sink sink) {
153 
154         sink.head();
155         sink.title();
156 
157         sink.text(bundle.getString("report.mlvp.header"));
158         sink.title_();
159         sink.head_();
160 
161         sink.body();
162 
163         sink.section1();
164         doGenerateLicenseCategories(bundle, sink);
165         sink.section1_();
166 
167         sink.section1();
168         doGenerateItemReport(bundle, sink);
169         sink.section1_();
170 
171 //The different scopes should be seen only if we have artifacts with an appropriately scope otherwise no headlines etc.
172         if (hasScope(Artifact.SCOPE_COMPILE)) {
173             sink.section1();
174             doGenerateArtifactCategories(bundle, sink, Artifact.SCOPE_COMPILE);
175             sink.section1_();
176         }
177 
178         if (hasScope(Artifact.SCOPE_TEST)) {
179             sink.section1();
180             doGenerateArtifactCategories(bundle, sink, Artifact.SCOPE_TEST);
181             sink.section1_();
182         }
183 
184         if (hasScope(Artifact.SCOPE_PROVIDED)) {
185             sink.section1();
186             doGenerateArtifactCategories(bundle, sink, Artifact.SCOPE_PROVIDED);
187             sink.section1_();
188         }
189 
190         if (hasScope(Artifact.SCOPE_RUNTIME)) {
191             sink.section1();
192             doGenerateArtifactCategories(bundle, sink, Artifact.SCOPE_RUNTIME);
193             sink.section1_();
194         }
195 
196         sink.body_();
197         sink.flush();
198         sink.close();
199     }

201     private boolean hasScope(String scope) {
202         boolean result = false;
203         for (LicenseInformation item : getLicenseInformations()) {
204             if (scope.equals(item.getArtifact().getScope())) {
205                 result = true;
206             }
207         }
208         return result;
209     }

271     private void doGenerateArtifactCategories(ResourceBundle bundle, Sink sink, String scope) {
272         sink.sectionTitle1();
273         sink.text("Artifact Categories (" + scope + ")");
274         sink.sectionTitle1_();
275 
276         if (hasValidEntries(scope)) {
277             doGenerateValidReport(bundle, sink, scope);
278         }
279         if (hasWarningEntries(scope)) {
280             doGenerateWarningReport(bundle, sink, scope);
281         }
282         if (hasInvalidEntries(scope)) {
283             doGenerateInvalidReport(bundle, sink, scope);
284         }
285         if (hasUnknownEntries(scope)) {
286             doGenerateUnknownReport(bundle, sink, scope);
287         }
288     }

290     private void doGenerateReportLicensesConfiguration(ResourceBundle bundle, Sink sink, String header, List<LicenseItem> licenseItems) {
291         sink.sectionTitle2();
292         sink.text(header);
293         sink.sectionTitle2_();
294 
295         sink.table();
296 
297         sink.tableRow();
298         headerCell(sink, "Id");
299         headerCell(sink, "Description");
300         headerCell(sink, "Names");
301         headerCell(sink, "URL's");
302         sink.tableRow_();
303 
304         for (LicenseItem item : licenseItems) {
305             sink.tableRow();
306             cell(sink, item.getId());
307             cell(sink, item.getDescription());
308 
309             sink.tableCell();
310             //List all names
311             sink.list();
312             for (String itemName : item.getNames()) {
313                 sink.listItem();
314                 sink.text(itemName);
315                 sink.listItem_();
316             }
317             sink.list_();
318             sink.tableCell_();
319 
320             sink.tableCell();
321             sink.list();
322             for (String itemURL : item.getUrls()) {
323                 sink.listItem();
324                 sink.link(itemURL);
325                 sink.text(itemURL);
326                 sink.link_();
327                 sink.listItem_();
328             }
329             sink.list_();
330             sink.tableCell_();
331 
332             sink.tableRow_();

354     private void doGenerateItemReport(ResourceBundle bundle, Sink sink) {
355         sink.sectionTitle1();
356         sink.text("Artifact License Categories");
357         sink.sectionTitle1_();
358 
359         sink.sectionTitle2();
360         sink.text("Overview");
361         sink.sectionTitle2_();
362 
363         sink.table();
364 
365         sink.tableRow();
366         headerCell(sink, "Id");
367         headerCell(sink, "Scope");
368         headerCell(sink, "Valid");
369         headerCell(sink, "Warning");
370         headerCell(sink, "Invalid");
371         headerCell(sink, "Unknown");
372         sink.tableRow_();
373 
374         for (LicenseInformation item : getLicenseInformations()) {
375             sink.tableRow();
376 
377             cell(sink, item.getArtifact().getId()); // 1st Row item artifactId
378 
379             boolean isValid = licenseValidator.isValid(item.getLicenses());
380             boolean isWarning = licenseValidator.isWarning(item.getLicenses());
381             boolean isInvalid = licenseValidator.isInvalid(item.getLicenses());
382             boolean isUnknown = licenseValidator.isUnknown(item.getLicenses());
383 
384             sink.tableCell();
385             sink.text(item.getArtifact().getScope());
386             sink.tableCell_();
387 
388             createFigure(sink, isValid);
389 
390             if (isWarning) {
391                 createFigure(sink, false);
392             } else {
393                 createEmptyCell(sink);
394             }
395 
396             if (isInvalid) {
397                 createFigure(sink, false);
398             } else {
399                 createEmptyCell(sink);
400             }
401 
402             if (isUnknown) {
403                 createFigure(sink, false);
404             } else {
405                 createEmptyCell(sink);
406             }
407 
408             sink.tableRow_();
409         }
410 
411         sink.table_();
412     }

605     public void generate(Sink sink, Locale locale) throws MavenReportException {
606         getLog().info("LicenseVerifierReport:generate()");
607         if (project.getArtifacts().isEmpty()) {
608             //If we have no artifacts at all.
609             doGenerateEmptyReport(getBundle(locale), sink);
610         } else {
611             try {
612                 loadLicensesFile();
613             } catch (MojoExecutionException e1) {
614                 throw new MavenReportException("Failure during load of the license.xml file", e1);
615             }
616 
617             //Get a set with all dependent artifacts incl.
618             //the transitive dependencies.
619             Set depArtifacts = project.getArtifacts();
620 
621             //Get all the informations about the licenses of the artifacts.
622             try {
623                 getDependArtifacts(depArtifacts);
624 
625                 //Sorting the artifacts...
626                 Collections.sort(getLicenseInformations(), new ArtifactComperator());
627                 doGenerateReport(getBundle(locale), sink);
628 
629             } catch (MojoExecutionException e) {
630                 getLog().error("Something wrong (BETTER MESSAGE): ", e);
631             }
632         }
633     }
