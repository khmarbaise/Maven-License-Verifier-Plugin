30  /**
31   * This goal is intended to check the artifacts during the prepare-package phase and
32   * print out warnings etc. based on the given configuration for the plugin.
33   *
34   * @goal verify
35   * @phase prepare-package
36   * @requiresDependencyResolution test
37   * @author <a href="mailto:kama@soebes.de">Karl Heinz Marbaise</a>
38   */
39  public class LicenseVerifierMojo
40      extends AbstractLicenseVerifierPlugIn
41  {
42  
43      public void execute()
44          throws MojoExecutionException
45      {
46  // The following code can be later used if we load a license.xml file from an
47  // different package.
48  //		ClassLoader sysClassLoader = this.getClass().getClassLoader();
49  //        URL[] urls = ((URLClassLoader)sysClassLoader).getURLs();
50  //        if (getLog().isDebugEnabled()) {
51  //			for (int i = 0; i < urls.length; i++) {
52  //				getLog().debug("ClassPath: " + urls[i].getFile());
53  //			}
54  //        }
55  //
56  //		try {
57  //			Enumeration<URL> list = this.getClass().getClassLoader().getResources("/license/*.xml");
58  //		} catch (IOException e) {
59  //			getLog().error("Exception: " + e.getMessage());
60  //		}
61          loadLicensesFile();
62  
63          //Get a set with all dependent artifacts incl.
64          //the transitive dependencies.
65          Set<?> depArtifacts = this.project.getArtifacts();
66  
67          //Check if we have to do something.
68          if (depArtifacts.isEmpty()) {
69              getLog().info("We haven't found any dependencies.");
70              return;
71          }
72  
73          //Get all the informations about the licenses of the artifacts.
74          getDependArtifacts(depArtifacts);
75  
76          Collections.sort(getLicenseInformations(), new ArtifactComperator());
77  
78          boolean isValid = false;
79          boolean isInvalid = false;
80          boolean isWarning = false;
81          boolean isUnknwon = false;
82  
83          for (LicenseInformation item : getLicenseInformations()) {
84  
85              /*
86               * [INFO]    XXXX
87               * [ERROR]   XXXX
88               * [WARNING] XXXX
89               *
90               * ]VALID[   XXX
91               * ]INVALID[ XXX
92               * ]WARNING[ XXX
93               * ]UNKNOWN[ XXX
94               */
95              if (licenseValidator.isValid(item.getLicenses())) {
96                  if (isVerbose()) {
97                      getLog().info("   ]VALID[   (" + item.getArtifact().getScope() + ") The artifact " + item.getProject().getId() + " has a license which is categorized as valid");
98                  }
99                  isValid = true;
100             } else if (licenseValidator.isInvalid(item.getLicenses())) {
101                 getLog().error("  ]INVALID[ (" + item.getArtifact().getScope() + ") The artifact " + item.getProject().getId() + " has a license which is categorized as invalid");
102                 isInvalid = true;
103             } else if (licenseValidator.isWarning(item.getLicenses())) {
104                 getLog().warn("]WARNING[ (" + item.getArtifact().getScope() + ") The artifact " + item.getProject().getId() + " has a license which is categorized as warning");
105                 isWarning = true;
106             } else if (licenseValidator.isUnknown(item.getLicenses())) {
107                 getLog().warn("]UNKNOWN[ (" + item.getArtifact().getScope() + ") The artifact " + item.getProject().getId() + " has a license which is categorized as unknown");
108                 isUnknwon = true;
109             }
110         }
111 
112         if (isValid && failOnValid) {
113             throw new MojoExecutionException("A license which is categorized as VALID has been found.");
114         }
115         if (isInvalid && failOnInvalid) {
116             throw new MojoExecutionException("A license which is categorized as INVALID has been found.");
117         }
118         if (isWarning && failOnWarning) {
119             throw new MojoExecutionException("A license which is categorized as WARNING has been found.");
120         }
121         if (isUnknwon && failOnUnknown) {
122             throw new MojoExecutionException("A license which is categorized as UNKNOWN has been found.");
123         }
124     }
