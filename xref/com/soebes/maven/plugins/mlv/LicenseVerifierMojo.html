24  /**
25   * This goal is intended to check the artifacts during the prepare-package phase and
26   * print out warnings etc. based on the given configuration for the plugin.
27   *
28   * @goal verify
29   * @phase prepare-package
30   * @requiresDependencyResolution test
31   * @author Karl Heinz Marbaise
32   */
33  public class LicenseVerifierMojo
34      extends AbstractLicenseVerifierPlugIn
35  {
36  
37      public void execute()
38          throws MojoExecutionException
39      {
40  // The following code can be later used if we load a license.xml file from an
41  // different package.
42  //		ClassLoader sysClassLoader = this.getClass().getClassLoader();
43  //        URL[] urls = ((URLClassLoader)sysClassLoader).getURLs();
44  //        if (getLog().isDebugEnabled()) {
45  //			for (int i = 0; i < urls.length; i++) {
46  //				getLog().debug("ClassPath: " + urls[i].getFile());
47  //			}
48  //        }
49  //
50  //		try {
51  //			Enumeration<URL> list = this.getClass().getClassLoader().getResources("/license/*.xml");
52  //		} catch (IOException e) {
53  //			getLog().error("Exception: " + e.getMessage());
54  //		}
55          loadLicensesFile();
56  
57          //Get a set with all dependent artifacts incl.
58          //the transitive dependencies.
59          Set<?> depArtifacts = this.project.getArtifacts();
60  
61          //Check if we have to do something.
62          if (depArtifacts.isEmpty()) {
63              getLog().info("We haven't found any dependencies.");
64              return;
65          }
66  
67          //Get all the informations about the licenses of the artifacts.
68          getDependArtifacts(depArtifacts);
69  
70          Collections.sort(getLicenseInformations(), new ArtifactComperator());
71  
72          boolean isValid = false;
73          boolean isInvalid = false;
74          boolean isWarning = false;
75          boolean isUnknwon = false;
76  
77          for (LicenseInformation item : getLicenseInformations()) {
78  
79              /*
80               * [INFO]    XXXX
81               * [ERROR]   XXXX
82               * [WARNING] XXXX
83               *
84               * ]VALID[   XXX
85               * ]INVALID[ XXX
86               * ]WARNING[ XXX
87               * ]UNKNOWN[ XXX
88               */
89              if (licenseValidator.isValid(item.getLicenses())) {
90                  if (isVerbose()) {
91                      getLog().info("   ]VALID[   (" + item.getArtifact().getScope() + ") The artifact " + item.getProject().getId() + " has a license which is categorized as valid");
92                  }
93                  isValid = true;
94              } else if (licenseValidator.isInvalid(item.getLicenses())) {
95                  getLog().error("  ]INVALID[ (" + item.getArtifact().getScope() + ") The artifact " + item.getProject().getId() + " has a license which is categorized as invalid");
96                  isInvalid = true;
97              } else if (licenseValidator.isWarning(item.getLicenses())) {
98                  getLog().warn("]WARNING[ (" + item.getArtifact().getScope() + ") The artifact " + item.getProject().getId() + " has a license which is categorized as warning");
99                  isWarning = true;
100             } else if (licenseValidator.isUnknown(item.getLicenses())) {
101                 getLog().warn("]UNKNOWN[ (" + item.getArtifact().getScope() + ") The artifact " + item.getProject().getId() + " has a license which is categorized as unknown");
102                 isUnknwon = true;
103             }
104         }
105 
106         if (isValid && failOnValid) {
107             throw new MojoExecutionException("A license which is categorized as VALID has been found.");
108         }
109         if (isInvalid && failOnInvalid) {
110             throw new MojoExecutionException("A license which is categorized as INVALID has been found.");
111         }
112         if (isWarning && failOnWarning) {
113             throw new MojoExecutionException("A license which is categorized as WARNING has been found.");
114         }
115         if (isUnknwon && failOnUnknown) {
116             throw new MojoExecutionException("A license which is categorized as UNKNOWN has been found.");
117         }
118     }
