22  package com.soebes.maven.plugins.mlv;

24  import java.util.Comparator;
25  
26  /**
27   * The comparator for the artifacts.
28   *
29   * @author <a href="mailto:kama@soebes.de">Karl Heinz Marbaise</a>
30   *
31   */
32  public class ArtifactComperator implements Comparator<LicenseInformation> {
33  
34      public int compare(LicenseInformation o1, LicenseInformation o2) {
35          int compareScope = o1.getArtifact().getScope().compareTo(o2.getArtifact().getScope());
36          if (compareScope == 0) {
37              return o1.getArtifact().getId().compareTo(o2.getArtifact().getId());
38          }
39          return compareScope;
40      }
41  
42  }
