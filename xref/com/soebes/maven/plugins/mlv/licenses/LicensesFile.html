16  package com.soebes.maven.plugins.mlv.licenses;
17  
18  import java.io.File;
19  import java.io.FileInputStream;
20  import java.io.IOException;
21  import java.io.StringWriter;
22  
23  import org.codehaus.plexus.util.xml.pull.XmlPullParserException;
24  
25  import com.soebes.maven.plugins.mlv.model.LicensesContainer;
26  import com.soebes.maven.plugins.mlv.model.io.xpp3.LicensesXpp3Reader;
27  import com.soebes.maven.plugins.mlv.model.io.xpp3.LicensesXpp3Writer;
28  
29  /**
30   * This class will serve methods for loading the licenses.xml file.
31   *
32   * @author Karl Heinz Marbaise
33   *
34   */
35  public class LicensesFile {
36  
37      /**
38       * Load the licenses.xml file and convert it into
39       * LicenseContainer class to have the information of the licenses.xml
40       * file available.
41       * @param licenseFile
42       * @return CheckLicenses instance.
43       * @throws XmlPullParserException
44       * @throws IOException
45       */
46      public static LicensesContainer getLicenses(File licenseFile) throws IOException, XmlPullParserException {
47          LicensesXpp3Reader read = new LicensesXpp3Reader();
48          LicensesContainer licenses = read.read(new FileInputStream(licenseFile));
49          return licenses;
50      }
51  
52      /**
53       * We convert the LicensesContainer into a string.
54       *
55       * @param licensesContainer
56       * @return
57       * @throws IOException
58       */
59      public static String toString(LicensesContainer licensesContainer) throws IOException {
60          StringWriter stringWriter = new StringWriter();
61          LicensesXpp3Writer xmlWriter = new LicensesXpp3Writer();
62          xmlWriter.write(stringWriter, licensesContainer);
63          return stringWriter.toString();
64      }
65  }
