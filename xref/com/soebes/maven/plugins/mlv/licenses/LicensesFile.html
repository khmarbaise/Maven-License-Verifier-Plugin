24  import java.io.File;

26  import java.io.IOException;
27  import java.io.InputStream;
28  import java.io.StringWriter;
29  
30  import org.codehaus.plexus.util.xml.pull.XmlPullParserException;
31  
32  import com.soebes.maven.plugins.mlv.model.LicensesContainer;
33  import com.soebes.maven.plugins.mlv.model.io.xpp3.LicensesXpp3Reader;
34  import com.soebes.maven.plugins.mlv.model.io.xpp3.LicensesXpp3Writer;
35  
36  /**
37   * This class will serve methods for loading the licenses.xml file.
38   *
39   * @author <a href="mailto:kama@soebes.de">Karl Heinz Marbaise</a>
40   *
41   */
42  public class LicensesFile {
43  
44      /**
45       * Load the licenses.xml file and convert it into
46       * LicenseContainer class to have the information of the licenses.xml
47       * file available.
48       * @param licenseFile
49       * @return LicensesContainer instance.
50       * @throws XmlPullParserException
51       * @throws IOException
52       */
53      public static LicensesContainer getLicenses(File licenseFile) throws IOException, XmlPullParserException {
54          return getLicenses(new FileInputStream(licenseFile));
55      }
56  
57      /**
58       * Load the licenses.xml file from an InputStream which
59       * usually means via getResourceAsStream().
60       *
61       * @param licensesInputStream The input stream where to load the license file from.
62       * @return LicensesContainer instance.
63       * @throws XmlPullParserException
64       * @throws IOException
65       */
66      public static LicensesContainer getLicenses(InputStream licensesInputStream) throws IOException, XmlPullParserException {
67          LicensesXpp3Reader read = new LicensesXpp3Reader();
68          LicensesContainer licenses = read.read(licensesInputStream);
69          return licenses;
70      }
71  
72      /**
73       * We convert the LicensesContainer into a string.
74       *
75       * @param licensesContainer
76       * @return
77       * @throws IOException
78       */
79      public static String toString(LicensesContainer licensesContainer) throws IOException {
80          StringWriter stringWriter = new StringWriter();
81          LicensesXpp3Writer xmlWriter = new LicensesXpp3Writer();
82          xmlWriter.write(stringWriter, licensesContainer);
83          return stringWriter.toString();
84      }
85  }
