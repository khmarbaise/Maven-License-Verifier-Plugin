144     /**
145      * This will check the given License against the given list of licenses.
146      *
147      * @param checkList
148      *            The list of licenses which will be used to check against.
149      * @param cl
150      *            The license which will be check to be part of the above list.
151      * @return true if the license has an equal name and/or equal URL false
152      *         otherwise.
153      */
154     private boolean check(List<LicenseItem> checkList, License cl) {
155         boolean result = false;
156         for (LicenseItem item : checkList) {
157             if (isStrictChecking()) {
158                 if (checkNamesAndURLs(item, cl)) {
159                     result = true;
160                 }
161             } else {
162                 if (checkNamesOrURLs(item, cl)) {
163                     result = true;
164                 }
165             }
166         }
167         return result;
168     }

170     private String checkId(List<LicenseItem> checkList, License cl) {
171         String result = null;
172         for (LicenseItem item : checkList) {
173             if (isStrictChecking()) {
174                 if (checkNamesAndURLs(item, cl)) {
175                     result = item.getId();
176                 }
177             } else {
178                 if (checkNamesOrURLs(item, cl)) {
179                     result = item.getId();
180                 }
181             }
182         }
183         return result;
184     }

337     /**
338      * TBD: Enhance description
339      *
340      * A list of licenses will be categorized as <code>Unknown</code> if one or
341      * more licenses can be categorized into different categories.
342      *
343      * For example if you have a list with three licenses
344      *
345      * isValid isWarning isInvalid isUnknown A X B X C X
346      *
347      * @param licenses
348      *            The list with the licenses which will be checked.
349      * @return true if a license is from no category.
350      *
351      */
352     public boolean isUnknown(ArrayList<License> licenses) {
353         boolean result = false;
354         if (licenses.isEmpty()) {
355             result = true;
356         } else {
357             int changeValid = 0;
358             int changeInvalid = 0;
359             int changeWarning = 0;
360             for (License license : licenses) {
361                 if (isValid(license)) {
362                     changeValid++;
363                 }
364                 if (isInvalid(license)) {
365                     changeInvalid++;
366                 }
367                 if (isWarning(license)) {
368                     changeWarning++;
369                 }
370             }
371 
372             if (changeValid == licenses.size()) {
373                 result = false;
374             } else if (changeInvalid == licenses.size()) {
375                 result = false;
376             } else if (changeWarning == licenses.size()) {
377                 result = false;
378             } else {
379                 result = true;
380             }
381         }
382         return result;
383     }
