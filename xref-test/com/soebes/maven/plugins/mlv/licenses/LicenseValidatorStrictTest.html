38  /**
39   * The intention of this test is to check if the comparsion only based on name
40   * and/or on URL of the license is working correctly with activated
41   * <b>strictChecking</b> option. That means both <b>name</b> and <b>url</b> must
42   * part of the licenses.xml file.

52      @BeforeClass
53      public void beforeClass() throws IOException, XmlPullParserException {
54          licensesContainer = LicensesFile.getLicenses(new File(getTestResourcesDirectory() + "licenses.xml"));
55          result = new LicenseValidator(licensesContainer);
56          result.setStrictChecking(true);

64      @Test
65      public void catagorizeCPLV10() {
66          License cl = new License();
67          cl.setName("Common Public License Version 1.0");
68          cl.setUrl("http://www.opensource.org/licenses/cpl1.0.txt");
69          Assert.assertTrue(result.isValid(cl));
70          Assert.assertFalse(result.isInvalid(cl));
71          Assert.assertFalse(result.isWarning(cl));
72          Assert.assertFalse(result.isUnknown(cl));
73      }
